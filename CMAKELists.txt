cmake_minimum_required(VERSION 3.16)
project(CephTrainer VERSION 2.0 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_FLAGS_DEBUG "-g -O0 -Wall -Wextra")
set(CMAKE_CXX_FLAGS_RELEASE "-O3 -DNDEBUG")

# Domyślnie Release jeśli nie podano
if(NOT CMAKE_BUILD_TYPE)
    set(CMAKE_BUILD_TYPE Release)
endif()

message(STATUS "Build type: ${CMAKE_BUILD_TYPE}")
message(STATUS "C++ Standard: ${CMAKE_CXX_STANDARD}")

find_package(OpenCV REQUIRED COMPONENTS core imgproc highgui imgcodecs)
message(STATUS "OpenCV found: ${OpenCV_VERSION}")
message(STATUS "OpenCV include: ${OpenCV_INCLUDE_DIRS}")

find_package(nlohmann_json QUIET)
if(NOT nlohmann_json_FOUND)
    find_path(NLOHMANN_JSON_INCLUDE_DIR
        NAMES nlohmann/json.hpp
        PATHS /usr/include /usr/local/include /opt/homebrew/include
    )
    if(NOT NLOHMANN_JSON_INCLUDE_DIR)
        message(FATAL_ERROR "nlohmann/json not found. Install via: sudo apt install nlohmann-json3-dev (Linux) or brew install nlohmann-json (macOS)")
    endif()
    message(STATUS "nlohmann/json include: ${NLOHMANN_JSON_INCLUDE_DIR}")
else()
    message(STATUS "nlohmann/json package found")
endif()

set(SOURCES
    src/main.cpp
    src/CephTrainer.cpp
    src/DataManager.cpp
    src/PatientLoader.cpp
    src/CephalometricAnalyzer.cpp
    src/Renderer.cpp
)

set(HEADERS
    include/PointTypes.h
    include/CephTrainer.h
    include/DataManager.h
    include/PatientLoader.h
    include/CephalometricAnalyzer.h
    include/Renderer.h
)

add_executable(CephTrainer ${SOURCES} ${HEADERS})

target_include_directories(CephTrainer PRIVATE
    ${CMAKE_CURRENT_SOURCE_DIR}/include
    ${OpenCV_INCLUDE_DIRS}
)

if(nlohmann_json_FOUND)
    target_link_libraries(CephTrainer PRIVATE
        ${OpenCV_LIBRARIES}
        nlohmann_json::nlohmann_json
    )
else()
    target_include_directories(CephTrainer PRIVATE ${NLOHMANN_JSON_INCLUDE_DIR})
    target_link_libraries(CephTrainer PRIVATE ${OpenCV_LIBRARIES})
endif()

if(MSVC)
    target_compile_options(CephTrainer PRIVATE /W4)
else()
    target_compile_options(CephTrainer PRIVATE -Wall -Wextra -Wpedantic)
    if(CMAKE_CXX_COMPILER_ID MATCHES "Clang")
        target_compile_options(CephTrainer PRIVATE -Wmost -Weverything -Wno-c++98-compat)
    endif()
endif()

install(TARGETS CephTrainer DESTINATION bin)

message(STATUS "───────────────────────────────────────────────────")
message(STATUS "CephTrainer configuration done!")
message(STATUS "Build with: cmake --build build --config ${CMAKE_BUILD_TYPE}")
message(STATUS "Run with:   ./build/CephTrainer")
message(STATUS "───────────────────────────────────────────────────")
